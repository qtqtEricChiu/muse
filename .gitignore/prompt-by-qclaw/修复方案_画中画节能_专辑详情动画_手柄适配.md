        pipFavBtn.textContent = isFaved ? '❤️' : '🤍';
        pipFavBtn.classList.toggle('faved', isFaved);
    }

    // 同步进度条
    const pipProgressFill = pipWindow.document.getElementById('pipProgressFill');
    if (pipProgressFill && audio.duration) {
        const pct = (audio.currentTime / audio.duration) * 100;
        pipProgressFill.style.width = `${pct}%`;
    }

    // ... 其余同步逻辑保持不变 ...
}, 1000);
